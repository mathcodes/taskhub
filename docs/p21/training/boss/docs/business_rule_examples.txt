﻿using System;
using System.Data;
using P21.Extensions.BusinessRule;

namespace P21.Extensions.Examples
{
    /*
     * Window: Order Entry
     * Setup - Order Tab:
     *  1.  Add user-defined field for discount percent
     *      name: discount_pct (system will name ufc_oe_hdr_ud_discount_pct)
     *      datetype: decimal
     *      decimal percision: 2
     *  2.  Add button
     *      name: system will name cb_usersd1 or something similar
     *      text: Add New Row for Discount
     *
     * Trigger: Order Header (Order tab) user defined button added above (cb_usersd1)
     * Rule Type: On Demand
     * Multi-Row:  Checked
     *
     * Fields passed to the rule:
     *		d_oe_header
     *			ufc_oe_hdr_ud_discount_pct (alias = discount_pct)
     *			cb_usersd1 (alias = add_discount_button)
     *		d_dw_oe_line_dataentry
     *			oe_order_item_id (alias = item_id)
     *			unit_price
     *		    unit_quantity
     *		    extended_price_total
     *
     * Description: Adds a new line item (other charge item) to the order to provide
     *              a flat percentage discount based on the order total.
     *
     *              Assumes there is an other charge item named "ORDER DISCOUNT" defined in the system
     *
     *              This rule will check if the "ORDER DISCOUNT" other charge item has already been added.
     *              - If it has not been added, the line item will be added with a quantity of 1 and unit price equal to the discount(negative).
     *              - If it has been added, the unit price will be updated to the current discount(negative).
     *
     */
    public class AddNewRowForDiscount : BusinessRule.Rule
    {

        public override RuleResult Execute()
        {
            //Item ID of other charge item that will be used to represent the discount
            const string discountItemID = "ORDER DISCOUNT";

            //  Get Order Header datatable from the data set
            DataTable orderHeader = Data.Set.Tables["d_oe_header"];
            //  Get Order Line datatable from the data set.
            DataTable orderLine = Data.Set.Tables["d_dw_oe_line_dataentry"];

            //  Get total order value from first line item
            decimal totalOrderValue;
            totalOrderValue = orderLine.Rows[0].Field<decimal>("extended_price_total");

            // Get discount percent entered by user
            decimal discountPct;
            discountPct = orderHeader.Rows[0].Field<decimal?>("discount_pct") ?? 0;

            //If discount was entered by user and order total is greater than 0, set discount by adding line or updating existing discount line
            if (totalOrderValue <= 0 || discountPct <= 0)
            {
                return RuleResult.SuccessfulResult;
            }
            DataRow discountRow;
            //Check if discount line already exists
            DataRow[] findRows = orderLine.Select("item_id = '" + discountItemID + "'");
            if (findRows.Length > 0)
            {
                discountRow = findRows[0];

                //Need to adjust order total to remove previous discount before recalculating
                totalOrderValue = totalOrderValue + (discountRow.Field<decimal>("unit_price") * -1);
            }
            else
            {
                //Add new row
                discountRow = Data.AddNewRow(orderLine);
                discountRow.SetField("item_id", discountItemID);
                //Quantity of 1.0 - discount will be determined via unit price
                discountRow.SetField("unit_quantity", Convert.ToDecimal(1.0));
            }

            //Calculate discount amount
            decimal discountAmt = totalOrderValue * (discountPct / 100);

            //Set unit price to discount amount (negative)
            discountRow.SetField("unit_price", discountAmt * -1);

            return RuleResult.SuccessfulResult;
        }
    }
}

    [RuleDescription("Branch Security", "Allow system to be secured by branch")]
    public class BranchSecurity : Rule
    {
        /*
		 * Purpose of the rule is to validate that the logged in user is secured to view the data
		 * at the branch associated with the location that has been passed into the rule.
		 *
		 * Demonstrates use of SQL within a business rule
		 *
		 */

        public override RuleResult Execute()
        {

            RuleResult result = new RuleResult();
            result.Success = false;     //assume failure, set to success if we get a successful retrieve on the security table
            result.Message = "You do not have the rights to review data from this branch."; //default message on failure

            //set up SQL connection, you'll need to use your own server / db. Integrated Security setting is for Windows Authentication.
            string sConn = "provider=sqloledb;server=(local)\\SQL2008;database=SalesEng;Integrated Security=SSPI";
            //using a table we created and added to our P21 database, it simply lists each user with the branches they are allowed
            //to view. the definition of the table is provided at the bottom
            string sSQL = @"select count(*) allowed_count
from users_x_branch
inner join location ON location.default_branch_id = users_x_branch.branch_id
where user_id = @userId
      AND location_id = @locationId";
            //create the database connection
            using var aConnection = new SqlConnection(sConn);

            //create the command object and store the sql query
            using var aCommand = new SqlCommand(sSQL, aConnection);
            aCommand.Parameters.Add(new SqlParameter("@userId", Data.Fields["global_user_id"].FieldValue));
            aCommand.Parameters.Add(new SqlParameter("@locationId", Data.Fields["location_id"].FieldValue));
            aConnection.Open();

            //create the datareader object to connect to table
            using var aReader = aCommand.ExecuteReader();
            aReader.Read();

            //if allowed_count > 0 then user is secured
            decimal allowedCount = 0;
            string recordsReturned = aReader["allowed_count"].ToString();
            if (recordsReturned != null && recordsReturned.Length > 0)
            {
                allowedCount = decimal.Parse(recordsReturned);
            }
            if (allowedCount > 0)
            {
                result.Success = true;      //user is allowed to see the data
                result.Message = "";
            }

            //close the reader
            aReader.Close();

            //close the connection -> important to do.
            aConnection.Close();

            return result;

﻿using System;
using System.Collections.Generic;
using System.Linq;
using P21.Extensions.BusinessRule;

namespace P21.Extensions.Examples.Rules
{
    /// <summary>
    /// I want to standardize country code to USA. Having US, United States,
    /// etc messes up my reports.
    /// </summary>
    [RuleDescription("Standardize Country Code", "Convert various names for USA to USA")]
    public class CountryHarmonizer : Rule
    {
        private readonly static string[] UsaSynonyms = new[]
        {
            "US",
            "U.S.",
            "U.S.A.",
            "United States",
            "United States of America"
        };

        public override RuleResult Execute()
        {
            //get the country from P21
            var field = Data.Fields.GetFieldByAlias("country");
            // Convert common synonyms for "United States" to "USA"
            if (UsaSynonyms.Contains(field.FieldValue))
            {
                //set to USA
                field.FieldValue = "USA";
            }
            //Default value for rule success is true.
            return RuleResult.SuccessfulResult;
        }
    }
}

﻿using System;
using System.Data;
using System.Data.SqlClient;
using System.Text;
using System.Xml.Linq;
using P21.Extensions.BusinessRule;

namespace P21.Extensions.Examples
{
    /*
     * Rule Type: On Event
     * Event: Form Datastream Created
     * Event Type: Invoices
     *
     * Description: This rule is fired when the form datastream is created for invoices. It will look up a
     *              user defined field for each item on the invoice and add it to the datastream so that
     *              information can be printed on the form. It will then also sort the invoice lines based
     *              on this new field.
     *
     *              For the purpose of this example, the user defined field is simply called invoice_sort_group.
     */
    [RuleDescription("Adds a user defined field to invoice form datastream lines and sorts the lines based on it.")]
    public class FormDatastream_SortInvoiceLines : BusinessRule.Rule
    {
        public override RuleResult Execute()
        {
            int invoiceSortGroup;
            string invoiceNo;
            string invoiceLineNo;
            DataTable invoiceSortGroupData;

            /*
             * During print preview, there will be multiple forms in the same datastream file. In this case we're going
             * to loop through each form since we're going to retrieve the user defined field per invoice. When physically
             * printing this will just loop once since there will be only 1 form per file. You could also get all lines in
             * the entire file using Data.XMLDatastream.GetLines().
             */
            try
            {
                foreach (XElement form in Data.XMLDatastream.GetForms())
                {
                    // Get the invoice number for the form and populate the DataTable
                    invoiceNo = Data.XMLDatastream.GetHeader(form).Element("INVOICE_NUMBER").Value;
                    invoiceSortGroupData = GetInvoiceSortGroupData(invoiceNo);

                    // Loop through each line for the form and get the invoice_sort_group for that item
                    foreach (XElement line in Data.XMLDatastream.GetLines(form))
                    {
                        invoiceLineNo = line.Element("LINE_NUMBER").Value;
                        invoiceSortGroup = 9999;

                        if (invoiceSortGroupData != null)
                        {
                            if (invoiceSortGroupData.Rows.Count > 0)
                            {
                                DataRow[] invoiceSortLine = invoiceSortGroupData.Select("invoice_no = '" + invoiceNo + "' AND line_no = " + invoiceLineNo);
                                if (invoiceSortLine.Length == 1)
                                {
                                    invoiceSortGroup = Convert.ToInt32(invoiceSortLine[0]["invoice_sort_group"]);
                                }
                            }
                        }

                        // Add the user defined field for invoice_sort_group to the invoice lines.
                        line.Add(new XElement("INVOICE_SORT_GROUP", invoiceSortGroup));
                    }
                }

                // Sort all lines by the invoice_sort_group. By not passing the form as the first parameter, this will sort all
                // forms by the given element. Also, pass true for the optional second parameter to indicate that we want to sort
                // in numeric order rather than alphabetic. Not specifying the optional 3rd parameter for sorting descending.
                Data.XMLDatastream.SortLines("INVOICE_SORT_GROUP", true);

                // Save the datastream back to the original file.
                Data.XMLDatastream.Document.Save(Data.XMLDatastream.FilePath);
            }
            catch (Exception ex)
            {
                Log.AddAndPersist(ex.Message);
            }

            return RuleResult.SuccessfulResult;
        }

        /// <summary>
        /// Returns a DataTable that is populated with the invoice sort group for each invoice line.
        /// </summary>
        private DataTable GetInvoiceSortGroupData(string invoiceNo)
        {
            DataTable dt = new DataTable();

            try
            {
                string sql = GetInvoiceSortGroupSql(invoiceNo);

                // We can't use the P21SqlConnection for rules triggered on the 'Form Datastream Created' event
                // since those are triggered from outside of P21. So we have to create our own connection here.
                string connectionString = string.Format(@"server={0};Database={1};User Id={2};Trusted_Connection=True;",
                                                        Session.Server,
                                                        Session.Database,
                                                        Session.UserID
                                                        );
                using (SqlConnection con = new SqlConnection(connectionString))
                {
                    con.Open();
                    using (SqlCommand cmd = new SqlCommand(sql, con))
                    {
                        using (SqlDataReader reader = cmd.ExecuteReader())
                        {
                            dt.Load(reader);
                            reader.Close();
                        }
                    }
                }
            }
            catch (Exception ex)
            {
                Log.AddAndPersist("Error retrieving invoice sort group info: " + ex.Message);
            }

            return dt;
        }

        /// <summary>
        /// Returns the SQL necessary to retrieve the user defined field "invoice_sort_group" for each line on the given invoice.
        /// </summary>
        private string GetInvoiceSortGroupSql(string invoiceNo)
        {
            StringBuilder invoiceSortGroupSql = new StringBuilder();

            invoiceSortGroupSql.AppendLine("SELECT	p21_view_invoice_line.invoice_no");
            invoiceSortGroupSql.AppendLine("      , p21_view_invoice_line.line_no");
            // If no invoice_sort_group exists for the line, give it 9999 so it sorts to the bottom.
            invoiceSortGroupSql.AppendLine("      , COALESCE(inv_mast_ud.invoice_sort_group, 9999) invoice_sort_group");
            invoiceSortGroupSql.AppendLine("FROM	p21_view_invoice_line ");
            invoiceSortGroupSql.AppendLine("LEFT JOIN inv_mast_ud ON inv_mast_ud.inv_mast_uid = p21_view_invoice_line.inv_mast_uid ");
            invoiceSortGroupSql.AppendLine("WHERE	p21_view_invoice_line.invoice_no = " + invoiceNo);

            return invoiceSortGroupSql.ToString();
        }

    }
